### 1. Supabase Vector Store Configuration (SQL)

We extend the initial schema with a **Weighted Similarity Function**. This allows the RAG pipeline to prioritize high-authority sources during the 120-second inference window.

```sql
-- 1. Optimize the Knowledge Base for Weighted Retrieval
-- We use HNSW for speed to stay under the 120s latency ceiling.
CREATE INDEX IF NOT EXISTS idx_knowledge_base_embedding 
ON knowledge_base 
USING hnsw (embedding vector_cosine_ops)
WITH (m = 16, ef_construction = 64);

-- 2. Create a Weighted Search Function
-- This function combines semantic similarity with the 'authority_weight'
-- Level 1 (0.95) sources will naturally float to the top even with slightly lower semantic match.
CREATE OR REPLACE FUNCTION weighted_knowledge_search (
  query_embedding vector(768), -- Adjusted for Gemini text-embedding-004 (768 dims)
  match_threshold float,
  match_count int,
  min_authority float
)
RETURNS TABLE (
  id uuid,
  content text,
  source_name text,
  authority_weight float,
  metadata jsonb,
  similarity float,
  weighted_score float
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT
    kb.id,
    kb.content,
    kb.source_name,
    kb.authority_weight,
    kb.metadata,
    1 - (kb.embedding <=> query_embedding) AS similarity,
    ((1 - (kb.embedding <=> query_embedding)) * kb.authority_weight) AS weighted_score
  FROM knowledge_base kb
  WHERE kb.authority_weight >= min_authority
    AND 1 - (kb.embedding <=> query_embedding) > match_threshold
  ORDER BY weighted_score DESC
  LIMIT match_count;
END;
$$;
```

### 2. Knowledge Moat Indexing Script (Python)

This script handles the ingestion of high-authority manuals. It uses `langchain` for orchestration and `google-generativeai` for state-of-the-art technical embeddings.

```python
import os
import hashlib
import json
from typing import List, Dict
from supabase import create_client, Client
import google.generativeai as genai
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import PyPDFLoader, UnstructuredHTMLLoader

# Configuration
SUPABASE_URL = os.environ.get("SUPABASE_URL")
SUPABASE_KEY = os.environ.get("SUPABASE_SERVICE_ROLE_KEY")
GEMINI_API_KEY = os.environ.get("GEMINI_API_KEY")

genai.configure(api_key=GEMINI_API_KEY)
supabase: Client = create_client(SUPABASE_URL, SUPABASE_KEY)

# Trusted Source Hierarchy Mapping
SOURCE_HIERARCHY = {
    "NHTSA": 0.95,
    "iFixit": 0.95,
    "OEM_MANUAL": 0.95,
    "REPAIR_CLINIC": 0.90,
    "REPAIR_PAL": 0.85,
    "TECH_FORUM": 0.65
}

class KnowledgeMoatIndexer:
    def __init__(self):
        # Technical-aware splitting: ensures steps and warnings stay together
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=1200,
            chunk_overlap=150,
            separators=["\n\n", "\n", "STEP ", "WARNING:", "CAUTION:", ". "]
        )

    def generate_checksum(self, content: str) -> str:
        """Creates the 'Immutable Safety Ledger' hash."""
        return hashlib.sha256(content.encode()).hexdigest()

    def get_embedding(self, text: str) -> List[float]:
        """Uses Gemini's latest embedding model."""
        result = genai.embed_content(
            model="models/text-embedding-004",
            content=text,
            task_type="retrieval_document"
        )
        return result['embedding']

    def process_document(self, file_path: str, source_type: str, metadata_ext: Dict):
        print(f"--- Processing: {file_path} ---")
        
        # 1. Load Document
        if file_path.endswith('.pdf'):
            loader = PyPDFLoader(file_path)
        else:
            loader = UnstructuredHTMLLoader(file_path)
        
        docs = loader.load()
        
        # 2. Chunking
        chunks = self.text_splitter.split_documents(docs)
        
        for chunk in chunks:
            # 3. Generate Checksum and Embedding
            checksum = self.generate_checksum(chunk.page_content)
            embedding = self.get_embedding(chunk.page_content)
            
            # 4. Prepare Metadata
            payload = {
                "content": chunk.page_content,
                "embedding": embedding,
                "source_name": source_type,
                "authority_weight": SOURCE_HIERARCHY.get(source_type, 0.5),
                "metadata": {
                    **metadata_ext,
                    "checksum": checksum,
                    "chunk_index": chunks.index(chunk),
                    "source_file": os.path.basename(file_path)
                }
            }
            
            # 5. Upsert to Supabase
            supabase.table("knowledge_base").upsert(payload).execute()
        
        print(f"Successfully indexed {len(chunks)} chunks from {source_type}.")

# Usage Example
if __name__ == "__main__":
    indexer = KnowledgeMoatIndexer()
    
    # Indexing an iFixit Manual (Level 1)
    indexer.process_document(
        file_path="./manuals/samsung_dryer_heating_element.pdf",
        source_type="iFixit",
        metadata_ext={"appliance": "Dryer", "brand": "Samsung", "model": "DV42H"}
    )
```

### 3. Retrieval Logic: The "Safety Gate" Query (Node.js)

This logic is used by the **Actionable Instruction Gate** to verify AI observations against the RAG knowledge moat before unlocking repair steps.

```typescript
// services/ragService.ts
import { createClient } from '@supabase/supabase-js';
import { GoogleGenerativeAI } from "@google/generative-ai";

const genai = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);
const supabase = createClient(process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_ROLE_KEY!);

export async function verifySafetyAgainstMoat(
  aiObservation: string, 
  minAuthority: number = 0.85
) {
  // 1. Generate query embedding for the observation
  const model = genai.getGenerativeModel({ model: "text-embedding-004" });
  const result = await model.embedContent(aiObservation);
  const embedding = result.embedding.values;

  // 2. Call the weighted search RPC
  const { data: matches, error } = await supabase.rpc('weighted_knowledge_search', {
    query_embedding: embedding,
    match_threshold: 0.7,
    match_count: 5,
    min_authority: minAuthority
  });

  if (error || !matches || matches.length === 0) {
    return { verified: false, reason: "No high-authority grounding found." };
  }

  // 3. Check for "Lethal" keywords in the top-weighted match
  const topMatch = matches[0];
  const lethalKeywords = ['high voltage', 'capacitor', 'gas line', 'pressurized'];
  const containsHazard = lethalKeywords.some(k => 
    topMatch.content.toLowerCase().includes(k)
  );

  return {
    verified: true,
    safety_checksum: topMatch.metadata.checksum,
    authority_score: topMatch.authority_weight,
    hazard_detected: containsHazard,
    grounding_text: topMatch.content,
    source: topMatch.source_name
  };
}
```

### Key Technical Decisions for this Subtask:
*   **Gemini `text-embedding-004`:** Selected for its 768-dimension efficiency, which reduces the vector calculation overhead by 50% compared to 1536-dim models, essential for the 120s latency limit.
*   **HNSW Indexing:** Chosen over IVFFlat to ensure that even as the "Trust Moat" grows to millions of manual pages, retrieval remains sub-second.
*   **Authority Multiplier:** Instead of a simple filter, the `weighted_score` allows the system to be "intelligent." If a Level 3 source has a 99% semantic match but a Level 1 source has a 90% match, the Level 1 source will still be prioritized for safety verification.
*   **Checksum Verification:** Every RAG result returned to the UI includes a `safety_checksum`, fulfilling the stakeholder requirement for an "Immutable Safety Ledger" for insurance/liability purposes.

### 1. Regex Kill-Switch: Stream Transformer Logic

To maintain the 120-second latency ceiling while ensuring safety, we use a `TransformStream`. This allows us to inspect the SSE chunks in real-time. ```typescript
```typescript
// lib/safety/KillSwitchTransformer.ts

/**
 * A comprehensive list of forbidden imperative verbs and phrases.
 * Categorized by risk: Mechanical, Electrical, and Chemical/Pressure.
 */
const FORBIDDEN_PATTERN = new RegExp(
  "\\b(" +
  // Mechanical actions
  "unscrew|loosen|tighten|remove|disassemble|pry|hammer|cut|drill|grind|" +
  // Electrical actions
  "touch|wire|solder|disconnect|reconnect|bridge|jump|short|test-lead|" +
  // Fluid/Pressure actions
  "drain|vent|bleed|refill|open-valve|pressurize|ignite|spray|" +
  // General Actionable Phrases
  "you should|make sure to|proceed by|start by|carefully" +
  ")\\b",
  "gi"
);

export class KillSwitchTransformer extends TransformStream {
  private isTerminated = false;

  constructor(onViolation: (violation: string) => void) {
    super({
      transform: (chunk, controller) => {
        if (this.isTerminated) return;

        const text = new TextDecoder().decode(chunk);
        
        // Check for forbidden imperatives
        const match = text.match(FORBIDDEN_PATTERN);
        
        if (match) {
          this.isTerminated = true;
          onViolation(match[0]);
          // Terminate the stream immediately to prevent further leakage
          controller.terminate();
          return;
        }

        controller.enqueue(chunk);
      }
    });
  }
}
```

```typescript
  verified: boolean;
  hazardLevel: 'Low' | 'Medium' | 'High' | 'Lethal';
  checksum: string;
  action: 'PROCEED' | 'BLOCK_AND_PIVOT';
  reasoning?: string;
}

export async function runSafetyFirewall(
  aiDiagnosis: any, 
  ragContext: any
): Promise<SafetyReport> {

  const components = aiDiagnosis.identified_components || [];

  const { data: hazards, error } = await supabaseAdmin
    .from('lethal_hazards_registry')
    .select('component_name, severity_score, mandatory_pivot')
    .in('component_name', components);

  // 3. Determine if any component triggers a "Lethal" flag
  const lethalTrigger = hazards?.find(h => h.severity_score >= 9 || h.mandatory_pivot);

  };

  report.checksum = generateSafetyChecksum(report as SafetyReport, aiDiagnosis.id);

  return report as SafetyReport;
}
```
